mysql命令行大全

Ⅰ 基础部分

  • -数据库启动————net start mysql;
  • -登录数据库————mysql -h '用户' -u '账号' -p '密码';
  • -新建数据库————create database [if not exists] '数据库名';
  • -选择数据库————use '数据库名'
  • -查看数据库————show databases;
  • -查看指定数据库——show database '数据库名';
  • -删除数据库————drop database [if exists] '数据库名';
  • -创建数据表————create table [if not exists] '表名'(
					'字段名'  '字段类型'  ,
					'字段名'  '字段类型' ,
					...
				);
  • -查看表结构————desc '数据表名';
  • -查看表创建语句——show create table '数据表名';
  • -查看表创建语句——show create table '数据表名';
  • -数据库停止——————net stop mysql;
